\newdimen\labeloffset
\newdimen\labelwidth

\labelwidth=5.5mm
\labeloffset=5mm

\def\beginlist{\vskip\parskip\begingroup\parskip=0pt
\dimen0=\labeloffset
\advance\dimen0 by \labelwidth
\parindent=\dimen0
\def\item[##1]{\par\hang\textindent{\hbox to \labelwidth{\hfil##1}}}
}
\def\endlist{\endgroup}

\def\hang{\hangindent\parindent}
\def\textindent#1{\indent\llap{#1\kern\labeloffset}\ignorespaces}
