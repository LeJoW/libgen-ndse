\def\dayTitle#1#2#3{\vfil\eject\cleardoublepage
\begingroup\display\fontdimen2\font=1.4\fontdimen2\font\baselineskip=40pt
\begincenterpar\strut15mm #1\endcenterpar\endgroup%
\vskip5mm
\global\lrunninghead={#3}\noheaderthispage}

\def\officeTitle#1#2{
\setbox0=\vbox{\begingroup\decorativ\it\baselineskip=17pt
	\begincenterpar\strut7mm\uppercase{#1}\endcenterpar
	\endgroup}%
\ifisroom\else\setendrule\fi
\bigskip\box0\bigskip\global\rrunninghead={#2}}

\def\sectionTitle#1{\noheaderthispage
	\centerline{\strut15mm\uppercase{#1}}
	\global\lrunninghead{#1}\bigskip}

\def\lessonTitle#1#2{\penalty-9999
	\medskip\def\detail{#2}
	\ifx\empty\detail
		\centerline{\strut7mm\decorativ #1}
	\else
		\line{\hfill\decorativ #1\hfill\hfill{\decorativ\it #2}\hfill}
	\fi\medskip}

\def\psalmTitle#1{\vskip0mm plus 1pt minus 1pt\centerline{\strut10mm\decorativ #1}}

\newif\ifenoughroom
\def\ifisroom{\updatefreespace
	\dimen0=\freespace
	\advance\dimen0 by -\bigskipamount
	\advance\dimen0 by -\ht0
	\advance\dimen0 by -\medskipamount
	\ifdim\dimen0>2cm \enoughroomtrue\else \enoughroomfalse\fi
	\ifenoughroom}

\def\setendrule{\ifdim\freespace>1cm\endrule\fi\eject}

\def\cleardoublepage{\ifodd\pageno\else\hbox{}\vfil\eject\fi}
