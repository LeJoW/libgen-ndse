\def\dayTitleShort{}
\def\officeTitleShort{}

\newdimen\titleSpaceLength
\titleSpaceLength=\fontdimen2\font

\newcommand{\titleSpace}{\hspace{\titleSpaceLength}}

\newcommand{\dayTitle}[3]{
	\thispagestyle{empty}
	\ifnum\thepage>1
		\cleardoublepage
	\fi
	\vspace*{2mm}
	{
		\titleSpaceLength=1.1em
		\thispagestyle{empty}
		{\centering \onehalfspacing \Huge\MakeUppercase{\decorativ #1}\par}
		\vspace{5mm}
		{\centering\it\decorativ\ttrad #2\par}
	}
	\global\def\dayTitleShort{#3}
}

\newcommand{\officeTitle}[2]{%
	\titleBox{%
		\vspace*{2mm}%
		\center{\uppercase{\it\decorativ #1}}%
		\vspace{2mm}%
	}%
	\global\def\officeTitleShort{#2}%
}

\def\center#1{%
{\centering #1\par}%
}

\newcommand{\sectionTitle}[1]{
	%\addvspace{5mm}
	{\centering\uppercase{#1}\par}
	%\addvspace{7mm}
}

\newcommand{\lessonTitle}[2]{
	\needspace{2cm}
	\vspace{4mm}
	\def\detail{#2}
	\ifx\empty\detail
		{\centering\decorativ #1\par}
		\vspace{2mm}
	\else
		{\hfill\decorativ#1\hfill\hfill{\it\decorativ #2}\hfill}
	\fi
}

\newcommand{\psalmTitle}[1]{%
	\needspace{2cm}%
	\addvspace{1.5mm}
	{\centering\decorativ #1\par}%
}

\def\titleBox#1{%
	\setbox0=\hbox to\textwidth{#1}%
	\updateSpaceLeft%
	\advance\spaceLeft by-\ht0%
	\ifdim\spaceLeft<2.4\gregoLineSkip%
		\widePageEnd%
		\penalty-10000%
	\fi%
	#1%
}

\def\widePageEnd{%
	\vfill%
	\hbox to \hsize{\hss\vrule width0.4\hsize height0.4pt depth0pt\hss}%
	\vfill%
}
