		\centerline{\decorativ #1}
		\vskip2mm
	\else
		\line{\hfill\decorativ #1\hfill\hfill{\decorativ\it #2}\hfill}
	\fi}

\def\ifisroom{\updatefreespace
	\dimen0=\freespace
	\advance\dimen0 by -\bigskipamount
	\advance\dimen0 by -\ht0
	\advance\dimen0 by -\medskipamount
	\ifdim\dimen0>2cm \enoughroomtrue\else \enoughroomfalse\fi
	\ifenoughroom}
